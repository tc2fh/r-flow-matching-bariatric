library(DBI)
library(odbc)
library(dplyr)

projects <- dbconnect (odbc(), .connection_string =
    "Driver={ODBC Driver 17 for SQL Server};
    Server=tcp: PROJECTS;
    Database=ProjectD332AFD;
    Trusted_Connection=yes;",
    timeout = 1000)

getqp <- function(sql, con=projects)
    dbGetQuery(statement = sql, conn = con)

glp1 <- getqp("SELECT * FROM GLP1Cohort WHERE Sex NOT IN (N'#Masked' , N'*Unspecified' ,N'Unknown', N'other')
                                             AND CoverageClass NOT IN (N'*Not Applicable', N'*Unspecified')
                                             AND BMIatEvent < WeightAtEvent
                                             AND BMIatEvent BETWEEN 35 AND 75
                                             AND (eGFRatEvent IS NULL OR eGFRatEvent >= 20)
                                             AND PMH_dialysis_transplant = 0
                                             AND (NephropathyInterval IS NULL OR NephropathyInterval >= 0)
                                             AND GLP1Name IN (N'semaglutide', N'tirzepatide', N'dulaglutide', N'liraglutide', N'exenatide microspheres')
                                             AND PMH_PriorMBS = 0
                                             AND MBSduringGLP1 = 0
                                             AND PMH_DM2 = 1
                                             AND PMH_retinopathy = 0
                                             AND ActiveEndInterval >= 700
                                             AND GLP1StartDate <= '2023-05-01'")

mbs <- getqp("SELECT * FROM MBSCohort WHERE Sex NOT IN (N'#Masked' , N' *Unspecified' ,N'Unknown' , N'other')
                                           AND CoverageClass NOT IN (N'*Not Applicable', N'*Unspecified')
                                           AND BMIatEvent < WeightAtEvent
                                           AND BMIatEvent BETWEEN 35 AND 75
                                           AND (eGFRatEvent IS NULL OR eGFRatEvent >= 20)
                                           AND PMH_dialysis_transplant = 0
                                           AND (NephropathyInterval IS NULL OR NephropathyInterval >= 0)
                                           AND PMH_PriorMBS = 0
                                           AND PriorGLP1 = 0
                                           AND PMH_DM2 = 1
                                           AND PMH_retinopathy = 0
                                           AND ActiveEndInterval >= 700
                                           AND ProcDatevalue <= '2023-05-01'")

dbDisconnect (projects)
